问题 A: 熟悉的A+B？

题目描述
A+B问题又来了。
设a，b，c是0到9之间的整数（其中a，b，c互不相同），其中abc和acc是两个不同的三位数，现给定一正整数n，问有多少对abc和acc能满足abc+acc=n？
输入
一个正整数n。
输出
第一行输出有多少对。
接下来每一行输出一对abc和acc，以空格分隔。如果没有一对abc和acc的话，则直接输出0即可。

样例输入

1068

样例输出

1
524 544

分析： 直接根据题目意思模拟……

代码

#include<bits/stdc++.h>
#define ll long long
using namespace std;
pair<int,int>cnt[10005];
int main(){
    int n;scanf("%d",&n);
    int ans=0;
    for(int a=1;a<=9;a++){
        for(int b=0;b<=9;b++){
            for(int c=0;c<=9;c++){
                if(a==b||a==c||b==c)continue;
                int abc=a*100+b*10+c;
                int acc=a*100+c*10+c;
                if(abc==acc)continue;
                if(abc+acc!=n)continue;
                cnt[ans++]={abc,acc};
            }
        }
    }
    cout<<ans<<endl;
    for(int i=0;i<ans;i++){
        cout<<cnt[i].first<<" "<<cnt[i].second<<endl;
    }return 0;
}

问题 B: 第n个数

题目描述
自从学了素数以后，明明喜欢上了数字2、3和5。当然，如果一个数字里面只出现2、3和5这三个数字，他也一样喜欢，例如222、2355、223355。
现在他希望你能够帮他编写一个程序，快速计算出由2、3、5这三个数字组成的由小到大的第n个数，当然也包括2、3和5。

输入
每组输入数据占1行，每行输入一个正整数n。（n<=1000）

输出
每组输出数据占1行，即满足要求的第n个数。

样例输入
3

样例输出
5

分析： 先将前1000个按顺序打表。2，3，5组成的由小到大的数很容易想到可以bfs，依次取2，3，5.

代码

#include <bits/stdc++.h>
#define ll long long
using namespace std;
int ans[1005];
int a[]={2,3,5};
void bfs(){
    queue<int>p;
    p.push(0);int k=0;
    while(!p.empty()){
        int t=p.front();p.pop();
        ans[k++]=t;
        if(k==1001)break;
        p.push(t*10+2);
        p.push(t*10+3);
        p.push(t*10+5);
    }
}
int main(){
    int n;bfs();
    while(~scanf("%d",&n)){
        cout<<ans[n]<<endl;
    }return 0;
}

问题 D: 小h的规划

题目描述
小h是一个很有规划的人，他列了很多事情，标注了优先级（1-9）也标注了必要程度（1-5）必要程度大的排前面，必要程度一样优先级大的排前面，必要程度一样，优先级一样不改变初始顺序。因为事情太多了，所以小h需要你帮忙..

输入
第一行一个数字n表示有n个事情
后n行每行两个数字a,b表示事情的优先级和必要程度
(n<=10000)

输出
将所有事情排序输出（只输出初始序号）

样例输入

2
9 3
1 4

样例输出

2 1

分析： 考察排序，使用C++的sort函数，重载排序规则。注意不改变初始顺序，可以使用stable_sort稳定的排序。或者重载的排序规则中，加上如果优先级和必要程度都相同，按照id从小到大排序。

代码

#include <bits/stdc++.h>
#define ll long long
#define INF 0x3f3f3f3f
using namespace std;
struct node{
    int a,b,id;
}s[10005];
bool cmp(const node &l,const node &r){
    if(l.b!=r.b)return l.b>r.b;
    else return l.a>r.a;
}
int main(){
    int n;scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d %d",&s[i].a,&s[i].b),s[i].id=i;
    stable_sort(s+1,s+1+n,cmp);
    for(int i=1;i<=n;i++)printf("%d ",s[i].id);
    printf("\n");
    return 0;
}

问题 E: 王子公主游戏

题目描述
小h最近发现一款游戏，在一个二维的地图中王子在（x1,y1）位置，公主在（x2,y2）位置。地图中有很多障碍物，判断王子是否能够救回公主（走到公主的位置）

输入
多组输入，第一行输入一个整数t，表示数据组数（1<=t<=10）
每组输入，第一行两个数字n和m表示地图的大小
后面n行，每行m个字符表示地图
S表示王子的位置，E表示公主的位置，’.’表示可以通过’#’表示障碍物
（n<=100,m<=100）

输出
对于每组数据判断王子是否能够救回公主,能的话输出YES否则输出NO

样例输入

2
2 2
.E
S.
2 2
#E
S#

样例输出

YES
NO

分析： 简单搜索，dfs或者bfs都可

代码

#include <bits/stdc++.h>
#define ll long long
#define INF 0x3f3f3f3f
using namespace std;
int n,m,sx,sy;
char mp[105][105];
typedef pair<int,int>PII;
int dis[105][105],vis[105][105];
int d[4][2]={1,0,-1,0,0,1,0,-1};
bool check(int x,int y){
    if(x<1||y<1||x>n||y>m)return 0;
    if(vis[x][y]||mp[x][y]=='#')return 0;
    return 1;
}
int bfs(){
    queue<PII>p;memset(dis,INF,sizeof(dis));
    memset(vis,0,sizeof(vis));dis[sx][sy]=0;
    p.push({sx,sy});
    while(!p.empty()){
        PII now=p.front();p.pop();
        int x=now.first,y=now.second;
        if(vis[x][y])continue;
        vis[x][y]=1;
        if(mp[x][y]=='E')return dis[x][y];
        for(int i=0;i<4;i++){
            int dx=x+d[i][0],dy=y+d[i][1];
            if(check(dx,dy)){
                if(dis[dx][dy]>dis[x][y]+1){
                    dis[dx][dy]=dis[x][y]+1;
                    p.push({dx,dy});
                }
            }
        }
    }return -1;
}
int main()
{
    int t;scanf("%d",&t);
    while(t--){
        scanf("%d %d",&n,&m);
        for(int i=1;i<=n;i++){
            scanf("%s",mp[i]+1);
            for(int j=1;j<=m;j++){
                if(mp[i][j]=='S')sx=i,sy=j;
            }
        }
        if(bfs()!=-1)cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }return 0;
}

问题 F: 小h去旅游

题目描述
小h想去p地旅游，p地有很多个旅游点，旅游点间有些有道路相连，有些没有。小h有若干询问，每个询问有两个数字，表示两个旅游点之间是否有道路相连，a->b->c也意味着a和c相连

输入
第一行三个数字n,m和q表示有n个旅游点（1到n），m条道路，q个查询
后m行每行两个数字u，v表示u，v两个旅游点有道路相连
后q行每行两个数字a，b查询a旅游点和b旅游点间是否有道路相连
(n<=10000,m<=10000,q<=10000)

输出
每个查询输出是否相连，是输出YES反之输出NO

样例输入

3 2 3
1 2
2 3
3 2
2 1
1 3

样例输出

YES
YES
YES

分析： 并查集，旅游点间是否可达，即判断两个旅游点之间是否在同一堆里面。

代码

#include <bits/stdc++.h>
#define ll long long
#define INF 0x3f3f3f3f

using namespace std;
int n,m,q;
int pre[10005];
int Find(int x){
    if(x!=pre[x])pre[x]=Find(pre[x]);
    return pre[x];
}
int main(){
    scanf("%d %d %d",&n,&m,&q);
    for(int i=1;i<=n;i++)pre[i]=i;
    while(m--){
        int u,v;scanf("%d %d",&u,&v);
        int fx=Find(u),fy=Find(v);
        if(fx!=fy)pre[fx]=fy;
    }
    while(q--){
        int a,b;scanf("%d %d",&a,&b);
        if(Find(a)==Find(b))cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }return 0;
}

问题 G: 最少数字

题目描述
小明用计算机随机生成了N个正整数，他希望从这N个数中选取若干个数，使得它们的和等于M。这些随机生成的数字可能会相同，但是每个数字最多只允许使用一次。
当然这样的选取方案可能不存在，也可能有多个。
现在希望你编写一个程序，能够找出数字个数最少的选取方案，输出对应的最少数字的个数，如果无解输出“No solution”。
输入
单组输入，每组输入2行。
第1行包含两个正整数N和M，分别表示初始输入的正整数个数和目标数字和(N<=1e3, M<=1e5)。
第2行为N个正整数，两两之间用空格隔开(每一个正整数均小于等于1e5)。

输出
输出数字个数最少的选取方案中所包含的最少数字个数，如果无解输出“No solution”。

样例输入

5 5
1 3 2 1 1

样例输出

2

代码

分析： 动态规划，状态转移方程：dp[i][j]=min(dp[i-1][j-a[i]]+1,dp[i-1][j]);(i代表第i个数字，j代表前i个数字组成的和)
压缩一维 dp[j]=min(dp[j],dp[j-a[i]]+1);

#include <bits/stdc++.h>
#define ll long long
#define INF 0x3f3f3f3f
using namespace std;
int a[1005],dp[100005];
int main(){
    int n,m;scanf("%d %d",&n,&m);memset(dp,INF,sizeof(dp));
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);dp[0]=0;
    for(int i=1;i<=n;i++){
        for(int j=m;j>=a[i];j--){
            dp[j]=min(dp[j],dp[j-a[i]]+1);
        }
    }
    if(dp[m]==INF)cout<<"No solution"<<endl;
    else cout<<dp[m]<<endl;
    return 0;
}

问题 H: Beautiful

题目描述
Mavis 有一个序列，对于每个数都有一个在序列中的优美值，这个优美值的定义是：找到序列中最长的一段，满足包含这个数并且这个数是这一段的中位数（以数值为第一关键字，下标为第二关键字排序, 这样的话这一段的长度只有可能是奇数），那么这一段的长度就是它的优美值。Mavis 说：“对于我每次手贱点出的左右端点 [l, r]，我都要找到 [l,r] 中的所有数中，最大的优美值”。
但是 Mavis 只会喊口号，不能解决问题，所以这个问题就交给你了。

输入
第一行输入 n，接下来 n 个整数，代表 ai。
接下来 Q，代表有 Q 个区间，接下来 Q 行，每行两个整数 l, r（l<=r），表示区间的左右端点。
1<=n<=2000,1<=Q<=100000,1<=ai<=200

输出
对于每个区间的询问，输出答案。

样例输入

8
16 19 7 8 9 11 20 16
8
3 8
1 4
2 3
1 1
5 5
1 2
2 8
7 8

样例输出

7
3
1
3
5
3
7
3

分析： 先看看优美值的定义，找到序列中最长的一段，满足包含这个数并且这个数是这一段的中位数，优美值就是这一段的长度。观察一下中位数：发现一段区间的中位数一定满足左边比它小的数的数量一定等于右边比他大的数的数量。

用两个数组记录一下向左和向右的每个上述数量的最长长度即可。
small和big数组中，自己赋值0，其余赋值−1
下面可以一边看代码一边理解。
枚举每一位数作为中位数，定义cnt为左边小于a[i]的数的个数（其实也不能严格说是个数，大于小于应该相互抵消，才能确定中位值，可结合代码理解）
为了方便操作，我们定义初始个数为n
这里讲一下当前位左边的操作，如果第j位小于等于第i位，那么cnt–，表示左边小于第i位的数多了一个。然后更新small数组，这样可以保证左边的小于a[i]的每一种个数的情况都是最长的。
同理右边cnt定义为大于a[i]的个数，后续操作也是一致的。

最后统计答案时，枚举[1−i,i−1]

代码

#include <bits/stdc++.h>
#define ll long long
#define INF 0x7f7f7f7f

using namespace std;
const int N=2005;
int a[N],small[N*2],big[N*2],val[N];
int n,q;
void solve(){
    for(int i=1;i<=n;i++){
        memset(small,-1,sizeof(small));
        memset(big,-1,sizeof(big));
        small[n]=big[n]=0;int cnt=n;
        for(int j=i-1;j>=1;j--){
            if(a[j]<=a[i])cnt--;
            else cnt++;
            small[cnt]=i-j;
        }cnt=n;
        for(int j=i+1;j<=n;j++){
            if(a[j]>=a[i])cnt++;
            else cnt--;
            big[cnt]=j-i;
        }
        for(int j=1-i;j<=i-1;j++){//从-（i-1）到（i-1）
            if(small[j+n]<0||big[n-j]<0)continue;
            val[i]=max(val[i],small[j+n]+big[n-j]+1);
        }
    }
}
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    solve();
    scanf("%d",&q);
    while(q--){
        int l,r;scanf("%d %d",&l,&r);
        int res=0;
        for(int i=l;i<=r;i++)res=max(res,val[i]);
        printf("%d\n",res);
    }return 0;
}

问题 I: Petrol

题目描述
花花所在的 C 国有 n 个城市，但是其中只有 k 个城市设有加油站，所有城市之间由 m 条双向道路连接，每条道路都有确定的长度。
花花想要买一辆新车，带上她家的 Doggie 去环游 C 国。但是车的油箱是有容量限制的，所以并不是所有的城市都能互相抵达，为了买上一辆心仪的车，花花会不断向你提问，询问在邮箱容量允许跑某个长度的情况下，某两个设有加油站的城市能否互相抵达。

输入
第一行包含 4 个由空格隔开的正整数 n、m、k、q。
接下来一行包含 k 个正整数，表示设有加油站的城市编号；
接下来 m 行，每行包含三个正整数 u、v、d，表示 u 城和 v 城之间有一条长度为 d 的双向道路连接；
接下来 q 行，每行包含三个正整数 u、v、d，表示花花想知道在油箱容量能允许跑 d 的长度时，u 城和 v 城能否互达。
n,q<=100000, m<=500000, k<=n

输出
输出共 q 行，每行包含一个字符串 “YES” 或者 “NO” ，表示花花询问的两个城市能否互达。

样例输入

6 5 4 4
1 5 2 6
1 3 1
2 3 2
3 4 3
4 5 5
6 4 5
1 2 4
2 6 9
1 5 9
6 5 8

样例输出

YES
YES
YES
NO

分析： dijkstra+kruskral+离线处理。
仔细分析题目，要求询问的两点之间，在距离d的限制下，是否可达。如果每一次都去跑一次dijkstra，肯定会T的。
那么我们继续往下看，题目只是要知道任意两个设有加油站的城市之间是否可达，那么其他无关点我们是否可以消去呢？答案是肯定的

如果从4号走到1号，直接沿简单路径走，油量最小为7，而如果从4到2再到1，就只需要5的油量就够了。4->2->1。所以做一次多源dijkstra，记录每一个点最近的加油站，然后如果一条边连接的两个点中最近的加油站不同，就新建一条连接着两个加油站的边，权值为dis[u]+dis[v]+w，dis为两个点到最近的加油站的距离，w为u，v之间的距离。
这样去除掉无关点之后，就会重新构建一个图，再对这个图求最小生成树，这里进行离线处理，将询问按照d从小到大排序，每一次将小于d的边全部连接起来，最后判断本次询问中的两个点，如果连接则输出yes

代码

#include <bits/stdc++.h>
#define INF 0x3f3f3f3f
using namespace std;
const int N=100005,M=500005*2;
int n,m,k,q;
int cnt,num;
int c[N],near[N],ans[N],pre[N];
int head[M],dis[N],vis[N];
typedef pair<int,int>PII;
struct query{
	int x,y,d,id;
	bool operator<(const query& a)const{
	    return d<a.d;
    }
}que[N];
struct node{
	int u,v,w;
	bool operator<(const node& a)const{
	    return w<a.w;
    }
}nod[M];
struct edge{
    int v,w,next;
}e[M];
void add(int u,int v,int w){
    e[cnt].v=v,e[cnt].next=head[u],e[cnt].w=w;head[u]=cnt++;
}
int Find(int x){
    if(x!=pre[x])pre[x]=Find(pre[x]);
    return pre[x];
}
void dijkstra(){//先记录每个点距离最近的加油站
	priority_queue<PII,vector<PII>,greater<PII> >p;
	for(int i=1;i<=n;i++)dis[i]=INF;
	for(int i=1;i<=k;i++){
		dis[c[i]]=0;
		near[c[i]]=c[i];
		p.push({0,c[i]});
	}
	while(!p.empty()){
		int u=p.top().second;p.pop();
		if(vis[u])continue;
		vis[u]=1;
		for(int i=head[u];~i;i=e[i].next){
            int v=e[i].v;
			if(dis[v]>dis[u]+e[i].w){
				dis[v]=dis[u]+e[i].w;
				near[v]=near[u];
				p.push({dis[v],v});
			}
		}
	}
}
void solve(){
	for(int i=1;i<=n;i++){//重构图
		for(int j=head[i];~j;j=e[j].next){
		    int v=e[j].v;
			if(near[i]!=near[v]){//相连两个点最近的加油站不同，将两个加油站直接连接
				nod[++num]=(node){near[i],near[v],dis[i]+dis[v]+e[j].w};
			}
		}
	}
	sort(nod+1,nod+1+num);
	for(int i=1;i<=n;i++)pre[i]=i;int now=1;
	for(int i=1;i<=q;i++){//离线处理
        while(now<=num&&nod[now].w<=que[i].d){
            int fx=Find(nod[now].u),fy=Find(nod[now].v);
            if(fx!=fy)pre[fx]=fy;
            now++;
        }
        int fx=Find(que[i].x),fy=Find(que[i].y);
        if(fx==fy)ans[que[i].id]=1;
        else ans[que[i].id]=0;
	}
	for(int i=1;i<=q;i++){
        if(ans[i])printf("YES\n");
        else printf("NO\n");
	}return ;
}

    memset(head,-1,sizeof(head));
    scanf("%d %d %d %d",&n,&m,&k,&q);
    for(int i=1;i<=k;i++)scanf("%d",&c[i]);
    for(int i=1;i<=m;i++){
        int u,v,d;scanf("%d %d %d",&u,&v,&d);
        add(u,v,d);add(v,u,d);
    }
    for(int i=1;i<=q;i++)scanf("%d %d %d",&que[i].x,&que[i].y,&que[i].d),que[i].id=i;
    sort(que+1,que+1+q);
    dijkstra();
    solve();
    return 0;
}
