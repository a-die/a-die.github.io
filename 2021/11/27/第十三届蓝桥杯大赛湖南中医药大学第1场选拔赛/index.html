B 三维迷宫

#include <bits/stdc++.h>
#define ll long long
using namespace std;
char s[1005];
int main()
{
    int x,y,z;
    scanf("%d %d %d",&x,&y,&z);
    scanf("%s",s);
    int n=strlen(s);
    for(int i=0;i<n;i++){
        if(s[i]=='E')x++;
        else if(s[i]=='W')x--;
        else if(s[i]=='S')y--;
        else if(s[i]=='N')y++;
        else if(s[i]=='U')z++;
        else if(s[i]=='D')z--;
    }
    printf("%d %d %d\n",x,y,z);
    return 0;
}

D 奖牌榜

#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#define ll long long
using namespace std;
const int N=10005;
struct node{
    int id,a,b,c,tot;
}e[N];
int cmp(const node &x,const node &y){
    if(x.a!=y.a)return x.a>y.a;
    else if(x.b!=y.b)return x.b>y.b;
    else if(x.c!=y.c)return x.c>y.c;
    else return x.id<y.id;
}
int main(){
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        int id,a,b,c;scanf("%d %d %d %d",&id,&a,&b,&c);
        e[i]={id,a,b,c,a+b+c};
    }
    sort(e+1,e+1+n,cmp);
    for(int i=1;i<=n;i++)printf("%d %d %d %d %d\n",e[i].id,e[i].a,e[i].b,e[i].c,e[i].tot);
    return 0;
}

E 两个集合

#include <bits/stdc++.h>
#define ll long long
using namespace std;
map<int,int>vis;
int a[10005];
int main()
{
    int m,n,k;
    scanf("%d %d %d",&m,&n,&k);
    for(int i=1;i<=m;i++){
        scanf("%d",&a[i]);
        vis[a[i]]++;
    }
    int res=0;
    for(int i=1;i<=n;i++){
        int b;scanf("%d",&b);
        res+=vis[k-b];
    }

}

F 数字盒子

#include <bits/stdc++.h>
#define ll long long
using namespace std;
int ans[1005],top;
string s;
int main()
{
    int n;scanf("%d",&n);
    for(int i=1;i<=n;i++){
        cin>>s;
        if(s=="Push")ans[++top]=1;
        else if(s=="Add"){
            if(top>0)ans[top]++;
        }
        else if(top>0)top--;
    }
    if(top==0)puts("No number.");
    else printf("%d\n",ans[top]);
    return 0;
}

I 低碳出行

#include <bits/stdc++.h>
#define ll long long
#define INF 0x3f3f3f3f
using namespace std;
const int N=1005,M=N*2;
int e[M],ne[M],t[M],w[M],head[N],tot;
int n,m;
int cw[N],time_[N],vis[N];
struct node{
    int cw,pos;
    bool operator <(const node &a) const{
        return cw>a.cw;
    }
};
void add(int u,int v,int c,int time){
    e[tot]=v,t[tot]=time,w[tot]=c,ne[tot]=head[u],head[u]=tot++;
}
void dijkstra(){
    priority_queue<node>p;
    p.push({0,1});
    memset(time_,0x3f,sizeof(time_));
    memset(cw,0x3f,sizeof(cw));
    time_[1]=0,cw[1]=0;
    while(!p.empty()){
        node now=p.top();
        p.pop();
        int u=now.pos;
        if(vis[u])continue;
        vis[u]=1;
        for(int i=head[u];~i;i=ne[i]){
            int v=e[i];
            if(cw[v]>cw[u]+w[i]){
                cw[v]=cw[u]+w[i];
                time_[v]=time_[u]+t[i];
                p.push({cw[v],v});
            }
            else if(cw[v]==cw[u]+w[i]&&time_[v]>time_[u]+t[i]){
                time_[v]=time_[u]+t[i];
            }
        }
    }
}
int main()
{
    scanf("%d %d",&n,&m);
    memset(head,-1,sizeof(head));
    for(int i=1;i<=m;i++){
        int u,v,c,T;
        scanf("%d %d %d %d",&u,&v,&c,&T);
        add(u,v,c,T),add(v,u,c,T);
    }
    dijkstra();
    printf("%d %d\n",cw[n],time_[n]);
    return 0;
}

J 小米的二叉树

#include <bits/stdc++.h>
#define ll long long
using namespace std;
const int N=1005;
int a[N],n;
int dfs(int rt){
    if(rt>n)return 0;
    int ans=0;
    ans+=dfs(rt*2)+a[rt];
    ans+=dfs(rt*2+1);
    return ans;
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    int v;scanf("%d",&v);
    int l=0,r=0;
    l+=dfs(v*2);
    r+=dfs(v*2+1);
    if(l==r)printf("E");
    else if(l>r)printf("L");
    else printf("R");
    return 0;
}
